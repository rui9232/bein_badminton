async function main() { try { await liff.init({ liffId: LIFF_ID }); if (!liff.isLoggedIn()) { liff.login(); return; } myProfile = await liff.getProfile(); // 診斷 1: 確認是否拿到 LINE ID console.log("LINE ID:", myProfile.userId); // 核心修正：使用 maybeSingle() 才不會因為找不到人而噴錯 let { data: profile, error: selectError } = await _supabase .from('profiles') .select('*') .eq('id', myProfile.userId) .maybeSingle(); if (selectError) { alert("查詢失敗: " + selectError.message); } // 如果資料庫沒這人，就執行建立 if (!profile) { console.log("正在為 芮 建立資料..."); const { data: newProfile, error: insertError } = await _supabase .from('profiles') .insert([{ id: myProfile.userId, display_name: myProfile.displayName, // 這裡必須對應截圖中的 display_name picture_url: myProfile.pictureUrl, role: 'user' }]) .select() .maybeSingle(); if (insertError) { // 診斷 2: 如果建立失敗，彈窗顯示原因（可能是 RLS 沒開好或欄位錯了） alert("建立資料失敗: " + insertError.message); } else { profile = newProfile; alert("芮，歡迎加入！已為您自動建立帳號。"); } } // 渲染畫面 document.getElementById('user-display').innerHTML = `
歡迎回來
${myProfile.displayName}
`; // 判斷權限 if (profile && (profile.role === 'admin' || profile.role === 'organizer')) { isAdmin = true; document.getElementById('admin-panel').style.display = 'block'; } loadEvents(); } catch (err) { alert("程式執行發生錯誤: " + err.message); console.error(err); } }
